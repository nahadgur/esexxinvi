# robots.txt — invisaligndentistsessex.uk
# Deliverable 5b: Crawl Budget Directives
#
# CRAWL BUDGET STRATEGY
# ─────────────────────────────────────────────────────────────────────────────
# The site has 783 indexable pages (666 matrix + 111 town hubs + 6 treatment
# hubs + ~10 static pages). This is a manageable crawl budget for a site of
# this domain age — but parameterized URLs from the directory search function,
# pagination, and tracking parameters can multiply the crawlable URL space by
# 10–50x if left unblocked.
#
# BLOCKING STRATEGY: Disallow query-string URL patterns at the path level.
# We cannot block individual query params directly in robots.txt — instead we
# disallow the paths that generate them. The canonical tag handles any that
# slip through.
#
# WHAT WE ARE BLOCKING
#   1. Search/filter parameters from the directory search function
#      (?q=, ?filter=, ?sort=, ?radius=, ?treatment=, ?tier=)
#   2. Pagination parameters (?page=, ?p=)
#   3. UTM and tracking parameters (?utm_*, ?ref=, ?source=, ?gclid=)
#   4. Session/auth parameters (?session=, ?token=, ?fbclid=)
#   5. Duplicate URL variants generated by case differences or index.html
#
# WHAT WE ARE NOT BLOCKING
#   - All clean /locations/[town]/ URLs (111 pages — indexable)
#   - All clean /locations/[town]/[service]/ URLs (666 pages — indexable)
#   - All /clinics/[slug]/ profile pages (indexable)
#   - All /treatments/[service]/ hub pages (indexable)
#   - /sitemap.xml and sub-sitemaps (must remain crawlable)
# ─────────────────────────────────────────────────────────────────────────────

User-agent: *

# ── Directory search function parameters ────────────────────────────────────
# The site search generates URLs like /locations/?q=chelmsford&filter=diamond
# These are functionally equivalent to the canonical /locations/ page.
Disallow: /*?q=
Disallow: /*?filter=
Disallow: /*?sort=
Disallow: /*?radius=
Disallow: /*?treatment=
Disallow: /*?tier=
Disallow: /*?type=
Disallow: /*?area=
Disallow: /*?near=
Disallow: /*?search=

# ── Pagination parameters ────────────────────────────────────────────────────
# Paginated results (if any) should use rel=prev/next, not param crawling.
Disallow: /*?page=
Disallow: /*?p=
Disallow: /*?pg=
Disallow: /*?offset=
Disallow: /*?start=
Disallow: /*?limit=

# ── UTM and marketing tracking parameters ───────────────────────────────────
# All UTM-tagged URLs are duplicates of their canonical equivalents.
Disallow: /*?utm_source=
Disallow: /*?utm_medium=
Disallow: /*?utm_campaign=
Disallow: /*?utm_term=
Disallow: /*?utm_content=
Disallow: /*?utm_id=

# ── Third-party click tracking and session parameters ───────────────────────
Disallow: /*?gclid=
Disallow: /*?fbclid=
Disallow: /*?msclkid=
Disallow: /*?ref=
Disallow: /*?source=
Disallow: /*?aff=
Disallow: /*?campaign=
Disallow: /*?session=
Disallow: /*?token=
Disallow: /*?sid=

# ── Print and modal variants ─────────────────────────────────────────────────
Disallow: /*?print=
Disallow: /*?modal=
Disallow: /*?popup=
Disallow: /*?lightbox=

# ── Next.js internal routes ─────────────────────────────────────────────────
# These should already return 404 but blocking them removes any crawl attempt.
Disallow: /_next/
Disallow: /api/

# ── Admin and form confirmation pages ───────────────────────────────────────
# noindex is also set on these via generateMetadata — double protection.
Disallow: /thank-you/
Disallow: /admin/

# ── Sitemaps — always allow ─────────────────────────────────────────────────
Allow: /sitemap.xml
Allow: /sitemap-static.xml
Allow: /sitemap-locations.xml
Allow: /sitemap-matrix.xml
Allow: /sitemap-clinics.xml
Allow: /sitemap-blog.xml

# ── Sitemap index declaration ────────────────────────────────────────────────
Sitemap: https://www.invisaligndentistsessex.uk/sitemap.xml
